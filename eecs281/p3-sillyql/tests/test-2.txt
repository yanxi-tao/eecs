CREATE items 2 string int category qty
INSERT INTO items 4 ROWS
food 10
tools 5
food 20
clothing 8
GENERATE FOR items hash INDEX ON category
INSERT INTO items 3 ROWS
food 15
tools 12
food 25
PRINT FROM items 1 qty WHERE category = food
PRINT FROM items 1 qty WHERE category = tools
PRINT FROM items 1 qty WHERE category = clothing
